알고리즘

프로그램 시작 
   맵 출력 
   입력 방향에 따라 캐릭터 이동시키기 
      W는 위로 이동 
      A는 왼쪽으로 이동 
      S는 아래로 이동 
      D는 오른쪽으로 이동 
   캐릭터의 다음 이동 방향 입력 받기 
프로그램 종료

프로그램 코드 

	// 파일명 : ./Chapter07/CharacterMovement.c

	#include <stdio.h>

	 

	// 프로그램 시작 

1	int main() {

		int row = 0, column = 9;

2		char direction='w';

	 

		// 맵 출력 

3		printf( " \033[44m                  \033[0m \n" );

4		while( ++row < 8 )

5			printf( " \033[44m  \033[0m                \033[44m  \033[0m \n" );

6		printf( " \033[44m                  \033[0m \n" );

		 

		// 입력 방향에 따라 캐릭터 이동시키기 

7		do {

			// 캐릭터의 기존 위치 지우고 새 위치로 업데이트

8			printf( " \033[%d;%df  ", row, column);

9			switch( direction ) {

				// W는 위로 이동 

10				case 'w': case 'W':

11					row = ( row <= 2 ) ? 2 : row-1;

12					break;

				// A는 왼쪽으로 이동 

13				case 'a': case 'A':

14					column = ( column <= 2 ) ? 2 : column-1;

15					break;

				// S는 아래로 이동 

16				case 's': case 'S':

17					row = ( row >= 8 ) ? 8 : row+1;

18					break;

				// D는 오른쪽으로 이동 

19				case 'd': case 'D':

20					column = ( column >= 15 ) ? 15 : column+1;

21					break;

			}

22			printf( " \033[%d;%df옷",row,column);

	 

			// 캐릭터의 다음 이동 방향 입력 받기 

23			printf( " \033[10;1f \033[2KWASD와 [Enter]를 입력하세요: " );

24			scanf( "\n%c", &direction);

25		} while( ( direction == 'W' ) || ( direction == 'A' ) || ( direction == 'S' ) || ( direction == 'D' )

		        || ( direction == 'w' ) || ( direction == 'a' ) || ( direction == 's' ) || ( direction == 'd' ) );

	 

		// 프로그램 종료 

26		return 0;

	}
