#include <stdio.h> 
#include <stdlib.h> 
#include <time.h> 
 
int main() { 
	srand( time( NULL ) ); 
	int number = rand() % 100 + 1; 
 
	int chance = 5; 
	( chance  0 ) { 
		printf( "1~100사이의 숫자를 입력하세요: " ); 
		int user = 0; 
		scanf( "%d", &user ); 
 
		 ( ( user  1 )  ( 100  user ) ) { 
			printf( "허용범위를 벗어났습니다\n" ); 
			chance++; 
			; 
		} 
		 ( number  user ) { 
			printf( "Success!!\n" ); 
			; 
		} 
		 ( number  user ) { 
			printf( "Up!\n" ); 
		} 
		 ( number  user ) { 
			printf( "Down!\n" ); 
		} 
		 
		printf( "기회는 %d번 남았습니다\n", chance ); 
	} 
 
	return 0; 
}
