코드 빈칸 채우기

#include <stdio.h>  
 
int main() { 
	int firstDay = 0; 
	int lastDay = 0; 
	int day = 0; 
	int dayOfWeek = 0; 
	int sunday = 0;		 
	int saturday = 6;		 
 
	printf( "일(0),월(1),화(2),수(3),목(4),금(5),토(6) 중에서 \n이번달 1일은 무슨요일인가요 : " ); 
	scanf("%d", &firstDay ); 
	printf( "이번달 말일은 언제인가요 : " ); 
	scanf("%d", &lastDay ); 
 
	( dayOfWeek = sunday; 										 
	                        dayOfWeek < firstDay;  
	                                             dayOfWeek   ) { 
		printf( "     " ); 
	} 
	 
	 ( day = 1; 									 
	              day  lastDay;  						 
	                             day  , dayOfWeek   ) { 
		printf( "", day );	 
		if ( dayOfWeek == saturday ) { 
			printf( "\n"); 
			dayOfWeek = sunday - 1; 
		} 
	} 
 
	return 0; 
}
