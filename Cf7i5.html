알고리즘

프로그램 시작 
   시작요일 및 말일 입력 받기 
   요일 출력 
   1일 이전은 공백 출력 
   1일부터 말일까지 출력 
      일주일을 한줄에 출력 
      즉, 토요일 후 일요일은 새 줄에서 시작 
프로그램 종료

프로그램 코드 

	// 파일명 : ./Chapter07/Calendar.c

	#include <stdio.h>

	 

	// 프로그램 시작 

1	int main() {

		int firstDay = 0;

		int lastDay = 0;

		int day = 0;

		int dayOfWeek = 0;

		int sunday = 0;

2		int saturday = 6;

	 

		// 시작요일 및 말일 입력 받기 

3		printf( "일(0),월(1),화(2),수(3),목(4),금(5),토(6) 중에서 \n이번달 1일은 무슨요일인가요 : " );

4		scanf("%d", &firstDay );

5		printf( "이번달 말일은 언제인가요 : " );

6		scanf("%d", &lastDay );

 

		// 요일 출력  

7		printf( "   일   월   화   수   목   금   토\n");

	 

		// 1일 이전은 공백 출력 

8		for( dayOfWeek = sunday;

9		                        dayOfWeek < firstDay;

10		                                             dayOfWeek++ ) {

11			printf( "     " );

		}

		 

		// 1일부터 말일까지 출력 

12		for ( day = 1;

13		              day <= lastDay;

14		                             day++, dayOfWeek++ ) {

			// 일주일을 한줄에 출력 

15			printf( "%5d", day );

			// 즉, 토요일 후 일요일은 새 줄에서 시작 

16			if ( ( dayOfWeek % 7 ) == saturday ) {

17				printf( "\n");

			}

		}

	 

		// 프로그램 종료 

18		return 0;

	}
