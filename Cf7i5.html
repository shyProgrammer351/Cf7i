알고리즘

프로그램 시작 
   임의의 수 숨기기 
   5번의 기회 내에서 임의의 수 맞추기 
      숫자 입력 받기 
         입력값이 허용범위 벗어나면 다시 입력 
      비교 결과 출력 
         첫째 임의의 수를 맞추면 Success!! 출력 후 종료 
         둘째 임의의 수가 입력값보다 크면 Up! 출력 
         셋째 임의의 수가 입력값보다 작으면 Down! 출력 
프로그램 종료

프로그램 코드 

	// 파일명 : ./Chapter07/UpAndDown.c

	#include <stdio.h>

	#include <stdlib.h>

	#include <time.h>

	 

	// 프로그램 시작 

1	int main() {

		// 임의의 수 숨기기 

		srand( time( NULL ) );

2		int number = rand() % 100 + 1;

	 

		// 5번의 기회 내에서 임의의 수 맞추기 

3		int chance = 5;

4		while( --chance >= 0 ) {

			// 숫자 입력 받기 

5			printf( "1~100사이의 숫자를 입력하세요: " );

			int user = 0;

6			scanf( "%d", &user );

	 

			// 입력값이 허용범위 벗어나면 다시 입력 

7			if ( ( user < 1 ) || ( 100 < user ) ) {

8				printf( "허용범위를 벗어났습니다\n" );

9				chance++;

10				continue;

			}

			// 비교 결과 출력 

			// 첫째 임의의 수를 맞추면 Success!! 출력 후 종료 

11			else if ( number == user ) {

12				printf( "Success!!\n" );

13				break;

			}

			// 둘째 임의의 수가 입력값보다 크면 Up! 출력 

14			else if ( number > user ) {

15				printf( "Up!\n" );

			}

			// 셋째 임의의 수가 입력값보다 작으면 Down! 출력 

16			else if ( number < user ) {

17				printf( "Down!\n" );

			}

			 

18			printf( "기회는 %d번 남았습니다\n", chance );

		}

	 

		// 프로그램 종료 

19		return 0;

	}
