알고리즘

프로그램 시작 
   사용자에게 엘리베이터에서 내릴 층수 입력받기 
   엘리베이터 문이 닫히면서 안내 메시지 출력 
   사용자가 1층부터 내릴 층수까지 엘리베이터가 한 층씩 이동하기 
      현재 층수 출력하고 한 층씩 올라가기 
   엘리베이터 문이 열리면서 도착 메시지 출력 
프로그램 종료

프로그램 코드 

	// 파일명 : ./Chapter04/Elevator.c 

	#include <stdio.h> 

 

	// 프로그램 시작 

1	int main() 

	{ 

		int currentFloor = 1;								 

2		int destinationFloor = 0; 

	 

		// 사용자에게 엘리베이터에서 내릴 층수 입력받기 

3		printf( "1층~5층 중 몇 층으로 올라가시나요 : " ); 

		// 2층을 입력한다고 가정 

4		scanf( "%d", &destinationFloor );  

	 

		// 엘리베이터 문이 닫히면서 안내 메시지 출력 

5		printf( "%d층으로 올라갑니다. 문이 닫힙니다.\n", destinationFloor ); 

	 

		// 사용자가 1층부터 내릴 층수까지 엘리베이터가 한 층씩 이동하기 

6		while( currentFloor <= destinationFloor )					 

		{ 

			// 현재 층수 출력하고 한 층씩 올라가기

7			printf( "%d↑\n", currentFloor ); 

8			currentFloor ++ ;							 

		} 

	 

		// 엘리베이터 문이 열리면서 도착 메시지 출력 

9		printf( "딩~동~댕~동~~ ♪~♬~♩~~ %d층입니다. 문이 열립니다", destinationFloor ); 

	 

		// 프로그램 종료 

10		return 0; 

	}
